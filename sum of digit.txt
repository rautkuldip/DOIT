import java.util.*;
import java.util.Scanner;
public class Main
{


	public static void main(String[] args) {
	    int num,digit;
	    int sum = 0;
	    Scanner sc = new Scanner(System.in);
	    num = sc.nextInt();
	    while(num > 0){
	         digit = num % 10;
	         sum = sum+digit;
	         num = num/10;
	    }
     System.out.println("sum of digit is:" + sum);   
	}
}
